float dx = 3; float dy = 2; float radius = 30; void setup() { size(400, 300); smooth(); } void draw() { background(240); fill(0, 150, 255); ellipse(x, y, radius*2, radius*2); x += dx; y += dy; if (x > width - radius || x < radius) dx *= -1; if (y > height - radius || y < radius) dy *= -1; }
